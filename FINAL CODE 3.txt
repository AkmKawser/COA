MOV CH,20D  ;FIXED INPUT

CMP CH ,030H  
JG  IFTRUE
JL  IFFALSEGOTOELSEIF  
JMP ELSEIFFALSEGOTOELSE
 
IFTRUE:
MOV AH,2
MOV DL,"N"
INT 21H
JMP EXIT

IFFALSEGOTOELSEIF:  
MOV AH,2
MOV DL,"P"
INT 21H
JMP EXIT


ELSEIFFALSEGOTOELSE:
MOV AH,2
MOV DL,"E"
INT 21H
JMP EXIT

EXIT:
MOV 4H,4CH
